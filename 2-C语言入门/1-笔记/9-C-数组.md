9.初始化实例:

A.全部初始化:

```
int a[5]={1,2,3,4,5};
int a[]={1,2,3,4,5}; // 合法,数组长度默认为5
int a[]={0}; //全零初始化
```

B.部分初始化:

```
int a[5]={1,2}; // 等价声明:
int a[5]={1,2,0,0,0};
```

C.没有初始化: 数组没有默认值,和没有初始化的x一样,都是垃圾值.

```
// 输入:
1.2 7 5 1.8 1.6 0.8 4 1.0 6 2.2
int main(void){
	float sum = 0;
	float average = 0;
	float a[10]={0};
	for(int i=0;i<10;i++){
		scanf("%f",&a[i]);
	}
	for(int i=0;i<10;i++){
		sum+=a[i];
	}
	average = sum / 10;
	float lessThanAverage[10]={0};
	int index = 0 ;
	for(int i=0;i<10;i++){
		if(a[i]<average){
			lessThanAverage[i]=a[index];
			index++;
		}	
	}
	// index为实际有效的index+1
	// 输出平均值
	printf("average is %f\n",average);
	// 输出小于平均值的元素
    for(int i=0;i<index;i++){
    	printf("Element is %f\n",lessThanAverage[i]);
    }
}


```
